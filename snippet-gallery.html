🔄 for 循环常见模式 循环入门W6

批量处理数据的核心武器。遍历列表、带序号遍历、并行遍历三大模式。

titles = ["经济向好", "科技突破", "体育赛事"]

# 模式1：直接遍历
for title in titles:
    print(title)

# 模式2：带序号遍历 enumerate（推荐！）
for i, title in enumerate(titles, 1):
    print(f"{i}. {title}")

# 模式3：遍历字典列表
news_data = [
    {"title": "新闻A", "views": 15000},
    {"title": "新闻B", "views": 85000},
]
for news in news_data:
    print(f"{news['title']} — {news['views']:,}阅读")

# 模式4：range() 数字循环
for i in range(5):        # 0,1,2,3,4
    print(i, end=" ")

# 模式5：zip() 并行遍历
names = ["人民日报", "新华社"]
fans = [15000000, 12000000]
for name, fan in zip(names, fans):
    print(f"{name}: {fan:,}粉丝")

🧩 函数定义与调用模板 函数入门W8

# ===== 基本函数 =====
def calc_avg_views(news_list):
    """计算新闻列表的平均阅读量"""
    if not news_list:
        return 0
    total = sum(n["views"] for n in news_list)
    return total / len(news_list)

# ===== 带默认参数 =====
def format_news(title, source, views, sep="｜"):
    return f"{title} {sep} {source} {sep} {views:,}阅读"

print(format_news("经济向好", "人民日报", 150000))

# ===== 多返回值 =====
def analyze(numbers):
    return min(numbers), max(numbers), sum(numbers)/len(numbers)

lo, hi, avg = analyze([15000, 85000, 230000])
print(f"最低:{lo:,} 最高:{hi:,} 平均:{avg:,.0f}")

🛡️ ETL 网络请求与伪装 数据采集中级W11

在抓取新闻网站或社交平台时，经常会遇到 403 Forbidden 错误。你需要使用 headers 来伪装成真实的浏览器。

import requests
import time
import random

# 定义高鲁棒性请求函数
def fetch_url(url):
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
        'Accept-Language': 'zh-CN,zh;q=0.9',
        'Referer': 'https://www.google.com/'
    }
    
    try:
        # 随机休眠 1-3 秒，防止被识别为机器人封禁
        time.sleep(random.uniform(1.0, 3.0))
        response = requests.get(url, headers=headers, timeout=10)
        response.raise_for_status() # 检查状态码是否为 200
        return response.text
    except Exception as e:
        print(f"请求失败: {e}")
        return None

🧹 正则表达式文本清洗 文本预处理W14

无论抓取的是微博还是新闻标题，往往掺杂了大量表情符号、HTML标签或无用标点。使用一键函数将其净化为纯净中文序列。

import re

def clean_text(text):
    if not isinstance(text, str):
        return ""
    # 1. 剔除 HTML 标签 (例如 <br>, <a> 等)
    text = re.sub(r'<[^>]+>', '', text)
    # 2. 剔除网页链接
    text = re.sub(r'http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\\(\\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+', '', text)
    # 3. 剔除特殊符号，仅保留中文字符、英文字母和数字
    # text = re.sub(r'[^\\u4e00-\\u9fa5a-zA-Z0-9]', '', text) 
    
    # 或者用 strip 去除前后空白
    return text.strip()

☁️ Jieba分词与极简词云 内容可视化W14

快速查看某一事件新闻标题或评论的核心议题分布。需提前安装 jieba 和 wordcloud。

import jieba
from wordcloud import WordCloud
import matplotlib.pyplot as plt

# 读取已合并的文本
text_data = "汕头大学学子在挑战杯中荣获佳绩..."

# 使用 jieba 精确模式分词
words = jieba.lcut(text_data)

# 过滤停用词与单字
stopwords = ['的', '了', '和', '是', '就', '在']
cleaned_words = [w for w in words if len(w) > 1 and w not in stopwords]

# 生成词云配置 (需确保有中文字体文件如 simhei.ttf)
wc = WordCloud(
    font_path='simhei.ttf', # 替换为你电脑上的中文字体路径
    width=800, 
    height=400,
    background_color='white',
    max_words=100
)

# 绘制
word_string = " ".join(cleaned_words)
wc.generate(word_string)

plt.figure(figsize=(10, 5))
plt.imshow(wc, interpolation='bilinear')
plt.axis('off')
plt.show()
